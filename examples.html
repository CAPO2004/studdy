$_GET Examples

1. Basic URL Parameter


<?php
$name = $_GET['name'];
echo "Welcome, " . $name;

?>

2. Check If Parameter Exists

<?php
if (isset($_GET['id'])) {
    $id = $_GET['id'];
    echo "Product ID: $id";
} else {
    echo "No ID provided";
}
?>

3. Multiple Parameters


<?php
$category = $_GET['category'];
$page = $_GET['page'];
$sort = $_GET['sort'];
echo "Showing $category sorted by $sort (Page $page)";

?>

4. Default Value If Missing

<?php
$page = $_GET['page'] ?? 1;
$limit = $_GET['limit'] ?? 10;
echo "Page: $page, Items per page: $limit";

?>

5. Sanitize Input (Security)

<?php
$search = $_GET['q'] ?? '';
$safe = htmlspecialchars($search, ENT_QUOTES, 'UTF-8');
echo "You searched for: $safe";

?>

6. Build Dynamic Links

<?php
$current = $_GET['page'] ?? 1;
$next = $current + 1;
$prev = max(1, $current - 1);
echo "<a href='?page=$prev'>Prev</a>";
echo " | Page $current | ";
echo "<a href='?page=$next'>Next</a>";
?>

$_POST Examples

1. Basic Form Data

<?php
$username = $_POST['username'];
$password = $_POST['password'];
echo "Welcome, $username";
?>

2. Check Request Method

<?php
if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $name = $_POST['name'];
    echo "Form submitted by: $name";
} else {
    echo "Please submit the form";
}
?>

3. Complete Contact Form

<?php
if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $name = htmlspecialchars($_POST['name']);
    $email = filter_var($_POST['email'], FILTER_SANITIZE_EMAIL);
    $message = htmlspecialchars($_POST['message']);
    echo "Name: $name<br>";
    echo "Email: $email<br>";
    echo "Message: $message";
}
?>

4. Validate Email

<?php
$email = $_POST['email'];
if (filter_var($email, FILTER_VALIDATE_EMAIL)) {
    echo "Valid email: $email";
} else {
    echo "Invalid email address";
}
?>

5. Password Hashing

<?php
$password = $_POST['password'];
$hashed = password_hash($password, PASSWORD_DEFAULT);
echo "Hashed: $hashed";

if (password_verify($password, $hashed)) {
    echo "Password correct!";
}
?>

6. Multiple Checkboxes


<?php
$skills = $_POST['skills'] ?? [];
foreach ($skills as $skill) {
    echo "Selected: $skill<br>";
}

?>

Arithmetic Examples

1. All Basic Operators

<?php
$a = 10;
$b = 3;
echo $a + $b;  
echo $a - $b;  
echo $a * $b;  
echo $a / $b;  
echo $a % $b;  
echo $a ** $b; 
?>

2. Calculate Total Price

<?php
$price = 99.99;
$quantity = 3;
$tax = 0.15;
$subtotal = $price * $quantity;
$total = $subtotal + ($subtotal * $tax);
echo "Total: $" . number_format($total, 2);

?>

3. Check Even/Odd (Modulus)

<?php
$num = 7;
if ($num % 2 == 0) {
    echo "$num is Even";
} else {
    echo "$num is Odd";
}

?>

4. Increment & Decrement

?>

5. Calculate Average

$sum = array_sum($scores);
$count = count($scores);
$average = $sum / $count;
echo "Average: $average";

?>

    echo "True";
}

Switch Examplesÿ£ŸÖÿ´ŸÑÿ© Switch

1. Basic Switch

<?php
$color = "Red";
switch ($color) {
    case "Red":
        echo "Is Red";
        break;
    case "Blue":
        echo "Is Blue";
        break;
}
?>

2. Default Case

<?php
switch ($day) {
    case "Sat":
        echo "Weekend";
        break;
    default:
        echo "Work day";
}
?>

3. Multiple Cases

<?php
switch ($fruit) {
    case "Apple":
    case "Pear":
        echo "Pome Fruit";
        break;
}
?>

4. Switch True

<?php
$age = 18;
switch (true) {
    case ($age < 10):
        echo "Child";
        break;
    case ($age >= 10):
        echo "Teen/Adult";
}
?>

5. Match (PHP 8)

<?php
$status = 200;
$msg = match($status) {
    200 => "OK",
    400 => "Error",
    default => "Unknown",
};
echo $msg;
?>

6. Alt Syntax

<?php
$i = 1;
switch ($i):
    case 1:
        echo "One";
        break;
endswitch;
?>

1. Count to 5

for ($i=1; $i<=5; $i++) {
    echo "Number: $i <br>";

    echo $i . "...";
}

    echo $i;
}

<?php
for ($x=1; $x<=2; $x++) {
    for ($y=1; $y<=2; $y++) {
        echo "$x,$y ";
    }
}
?>

While Loop Examplesÿ£ŸÖÿ´ŸÑÿ© ÿ≠ŸÑŸÇÿ© While

1. Basic While

<?php
$x = 1;
while ($x <= 5) {
    echo $x++;
}
?>

2. Do...While

<?php
$x = 10;
do {
    echo "Runs once";
} while ($x < 5);
?>

3. Break Condition

<?php

while (true) {
    $i++;
    if ($i > 10) break;
}
?>

4. While Continue

<?php
$i = 0;
while ($i < 5) {
    $i++;
    if ($i == 3) continue;
    echo $i;
}
?>

5. Random Until Match

<?php
$n = 0;
while ($n != 5) {
    $n = rand(1, 10);
    echo $n . " ";
}
?>

6. Nested While

<?php
$i = 1;
while ($i <= 2) {
    $j = 1;
    while ($j <= 2) {
        echo "$i-$j ";
        $j++;
    }
    $i++;
}
?>
