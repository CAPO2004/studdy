$_GET Examples

1. Basic URL Variable

Receiving a name from the URL (e.g., page.php?name=Ahmed).

<?php
$name = $_GET['name'];
echo "Welcome, " . $name;
?>

2. Handling Missing Data

Using isset() to prevent errors if the parameter is missing.

<?php
if (isset($_GET['id'])) {
    echo "Product ID: " . $_GET['id'];
} else {
    echo "No ID provided.";
}
?>

3. Multiple Parameters

Reading multiple values (e.g., ?category=books&page=2).

<?php
$cat = $_GET['category'];
$page = $_GET['page'];
echo "Category: $cat, Page: $page";
?>

$_POST Examples

1. Simple Login Form

<?php
$user = $_POST['username'];
$pass = $_POST['password'];
echo "Logging in user: " . $user;
?>

2. Checking Request Method

<?php
if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    echo "Form submitted!";
}
?>

3. Secure Input

<?php
$email = htmlspecialchars($_POST['email']);
echo "Safe email: " . $email;
?>

Arithmetic Examples

1. Basic Calculation

<?php
$total = 100 + 50;
echo "Total: " . $total; // 150
?>

2. Modulus (Remainder)

<?php
$rem = 10 % 3;
echo "Remainder: " . $rem; // 1
?>

3. Increment Operator

<?php
$count = 0;
$count++;
echo $count; // 1
?>

Comparison Examples

1. Equal (==)

<?php
if (5 == "5") {
    echo "True: Values are equal.";
}
?>

2. Identical (===)

<?php
if (5 === "5") {
    echo "False"; // Different types
}
?>

3. Greater/Less Than

<?php
$age = 20;
if ($age >= 18) {
    echo "Adult";
}
?>

If/Else Examples

1. Simple If

<?php
$score = 90;
if ($score > 50) {
    echo "Pass";
}
?>

2. If...Else

<?php
$light = "Red";
if ($light == "Green") {
    echo "Go";
} else {
    echo "Stop";
}
?>

3. Elseif Chain

<?php
$t = 15;
if ($t < 10) {
    echo "Morning";
} elseif ($t < 20) {
    echo "Day";
} else {
    echo "Night";
}
?>

Switch Examples

1. Basic Switch

<?php
$color = "Red";
switch ($color) {
    case "Red":
        echo "Is Red";
        break;
    case "Blue":
        echo "Is Blue";
        break;
}
?>

2. Default Case

<?php
switch ($day) {
    case "Sat":
        echo "Weekend";
        break;
    default:
        echo "Work day";
}
?>

3. Multiple Cases

<?php
switch ($fruit) {
    case "Apple":
    case "Pear":
        echo "Pome Fruit";
        break;
}
?>

For Loop Examples

1. Count to 5

<?php
for ($i=1; $i<=5; $i++) {
    echo "Valid: $i 
";
}
?>

2. Count Down

<?php
for ($i=10; $i>0; $i--) {
    echo $i . "...";
}
?>

3. Array Loop

<?php
$colors = ["Red", "Blue"];
for ($i=0; $i < count($colors); $i++) {
    echo $colors[$i];
}
?>

While Loop Examples

1. Basic While

<?php
$x = 1;
while ($x <= 5) {
    echo $x++;
}
?>

2. Do...While

<?php
$x = 10;
do {
    echo "Runs once";
} while ($x < 5);
?>

3. Break Condition

<?php
$i = 0;
while (true) {
    $i++;
    if ($i > 10) break;
}
?>
