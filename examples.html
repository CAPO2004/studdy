$_GET Examples

1. Basic URL Parameter

// URL: page.php?name=Ahmed
<?php
$name = $_GET['name'];
echo "Welcome, " . $name;
// Output: Welcome, Ahmed
?>

2. Check If Parameter Exists

<?php
if (isset($_GET['id'])) {
    $id = $_GET['id'];
    echo "Product ID: $id";
} else {
    echo "No ID provided";
}
?>

3. Multiple Parameters

// URL: shop.php?category=electronics&page=2&sort=price
<?php
$category = $_GET['category'];
$page = $_GET['page'];
$sort = $_GET['sort'];
echo "Showing $category sorted by $sort (Page $page)";
// Output: Showing electronics sorted by price (Page 2)
?>

4. Default Value If Missing

<?php
$page = $_GET['page'] ?? 1;
$limit = $_GET['limit'] ?? 10;
echo "Page: $page, Items per page: $limit";
// If no params: Page: 1, Items per page: 10
?>

5. Sanitize Input (Security)

<?php
$search = $_GET['q'] ?? '';
$safe = htmlspecialchars($search, ENT_QUOTES, 'UTF-8');
echo "You searched for: $safe";
// Prevents XSS attacks
?>

6. Build Dynamic Links

<?php
$current = $_GET['page'] ?? 1;
$next = $current + 1;
$prev = max(1, $current - 1);
echo "<a href='?page=$prev'>Prev</a>";
echo " | Page $current | ";
echo "<a href='?page=$next'>Next</a>";
?>

$_POST Examples

1. Basic Form Data

<?php
$username = $_POST['username'];
$password = $_POST['password'];
echo "Welcome, $username";
?>

2. Check Request Method

<?php
if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $name = $_POST['name'];
    echo "Form submitted by: $name";
} else {
    echo "Please submit the form";
}
?>

3. Complete Contact Form

<?php
if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $name = htmlspecialchars($_POST['name']);
    $email = filter_var($_POST['email'], FILTER_SANITIZE_EMAIL);
    $message = htmlspecialchars($_POST['message']);
    
    echo "Name: $name<br>";
    echo "Email: $email<br>";
    echo "Message: $message";
}
?>

4. Validate Email

<?php
$email = $_POST['email'];
if (filter_var($email, FILTER_VALIDATE_EMAIL)) {
    echo "Valid email: $email";
} else {
    echo "Invalid email address";
}
?>

5. Password Hashing

<?php
$password = $_POST['password'];
$hashed = password_hash($password, PASSWORD_DEFAULT);
echo "Hashed: $hashed";

// To verify later:
if (password_verify($password, $hashed)) {
    echo "Password correct!";
}
?>

6. Multiple Checkboxes

// HTML: <input type="checkbox" name="skills[]" value="PHP">
<?php
$skills = $_POST['skills'] ?? [];
foreach ($skills as $skill) {
    echo "Selected: $skill<br>";
}
// Output: Selected: PHP, Selected: JS, etc.
?>

Arithmetic Examples

1. All Basic Operators

<?php
$a = 10;
$b = 3;

echo $a + $b;  // Addition: 13
echo $a - $b;  // Subtraction: 7
echo $a * $b;  // Multiplication: 30
echo $a / $b;  // Division: 3.333...
echo $a % $b;  // Modulus: 1
echo $a ** $b; // Exponent: 1000
?>

2. Calculate Total Price

<?php
$price = 99.99;
$quantity = 3;
$tax = 0.15;

$subtotal = $price * $quantity;
$total = $subtotal + ($subtotal * $tax);
echo "Total: $" . number_format($total, 2);
// Output: Total: $344.97
?>

3. Check Even/Odd (Modulus)

<?php
$num = 7;

if ($num % 2 == 0) {
    echo "$num is Even";
} else {
    echo "$num is Odd";
}
// Output: 7 is Odd
?>

4. Increment & Decrement

<?php
$x = 5;

echo ++$x; // Pre-increment: 6
echo $x++; // Post-increment: 6 (then x=7)
echo --$x; // Pre-decrement: 6
echo $x--; // Post-decrement: 6 (then x=5)
?>

5. Calculate Average

<?php
$scores = [85, 90, 78, 92, 88];
$sum = array_sum($scores);
$count = count($scores);
$average = $sum / $count;

echo "Average: $average";
// Output: Average: 86.6
?>

6. Compound Assignment

<?php
$x = 10;

$x += 5;  // Same as: $x = $x + 5 ‚Üí 15
$x -= 3;  // Same as: $x = $x - 3 ‚Üí 12
$x *= 2;  // Same as: $x = $x * 2 ‚Üí 24
$x /= 4;  // Same as: $x = $x / 4 ‚Üí 6

echo $x; // 6
?>
